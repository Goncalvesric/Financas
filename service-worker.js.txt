const CACHE_NAME = 'finance-tracker-cache-v1';
const urlsToCache = ['/', '/index.html'];
self.addEventListener('install', event => { event.waitUntil(caches.open(CACHE_NAME).then(cache => cache.addAll(urlsToCache))); });
self.addEventListener('fetch', event => { if (event.request.method !== 'GET' || event.request.url.includes('firestore.googleapis.com')) { return; } event.respondWith(caches.match(event.request).then(response => response || fetch(event.request))); });
